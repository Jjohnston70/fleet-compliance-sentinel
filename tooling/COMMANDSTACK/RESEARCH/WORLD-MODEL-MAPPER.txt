async function checkCFRFreshness() {
  const sources = [
    { id: "cfr-part-382", url: "https://www.ecfr.gov/current/title-49/part-382" },
    { id: "cfr-part-391", url: "https://www.ecfr.gov/current/title-49/part-391" }
  ];
  
  for (const source of sources) {
    const currentHash = await fetchAndHash(source.url);
    const storedHash = await db.knowledgeFreshness.findUnique({
      where: { source_id: source.id }
    });
    
    if (!storedHash || currentHash !== storedHash.hash) {
      // CFR changed!
      await db.knowledgeFreshness.upsert({
        where: { source_id: source.id },
        update: { 
          hash: currentHash, 
          status: "stale",
          last_changed: new Date() 
        },
        create: { source_id: source.id, hash: currentHash }
      });
      
      // Alert admin
      await sendEmail({
        to: "jacob@truenorthstrategyops.com",
        subject: `CFR Update Detected: ${source.id}`,
        body: `${source.url} changed. Re-index needed.`
      });
      
      // Warn users in Penny responses
      await db.knowledgeWarnings.create({
        data: {
          source_id: source.id,
          message: "This regulation may have changed recently. Verify before acting.",
          expires_at: addDays(new Date(), 90)
        }
      });
    }
  }
}

Gap Fix #2: Citation Verification
Production Fix (Immediate):
typescript// Add post-generation validation to Penny pipeline
async function validateCitations(response: string, tenantId: string) {
  // Extract citations from response
  const citations = response.match(/§\s*\d+\.\d+/g) || [];
  const validationResults = [];
  
  for (const citation of citations) {
    // Query vector store for cited section
    const chunks = await vectorStore.search({
      query: `section ${citation}`,
      tenantId,
      topK: 3
    });
    
    // Check if citation exists in knowledge base
    const found = chunks.some(chunk => 
      chunk.metadata.section_id === citation
    );
    
    if (!found) {
      validationResults.push({
        citation,
        status: "unverified",
        confidence: 0.0,
        warning: "Citation not found in knowledge base"
      });
    } else {
      // Compute confidence based on chunk similarity
      const topChunk = chunks[0];
      validationResults.push({
        citation,
        status: "verified",
        confidence: topChunk.score,
        source: topChunk.metadata.source_file
      });
    }
  }
  
  // Append warnings for unverified citations
  const unverified = validationResults.filter(r => r.status === "unverified");
  if (unverified.length > 0) {
    response += "\n\n⚠️ **Citation Warning:** Some cited regulations could not be verified in our knowledge base. Please verify before relying on this information.";
  }
  
  return { response, validationResults };
}

Impact: HIGH (prevents hallucinated citations)
